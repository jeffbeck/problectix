\begin{aufgabe}[Technologie]{Messfehler}
\begin{teilaufgabe}[ohnenummer]{t}{0}{2}
    Kreuzen Sie an, bei welchen Fehlern es sich um \textit{Systematische
      Messabweichungen} handelt. Falsch angekreuzte Antworten ergeben
    Punktabzug!
\begin{multicols}{2}
\begin{mch}[3]
  \item Abgenutzte Messflächen am Messschieber.
  \item Grat am Werkstück nicht entfernt
  \item Schräges Ansetzen des Messschiebers (Kippfehler)
  \item Ablesefehler durch Parallaxe
  \item Formänderung des Werkstücks durch ungleichmäßige Messkraft
  \item Abweichende mm-Teilung an der Skala eines Messschiebers
\end{mch}
\end{multicols}
\korrektur{-9mm}
\end{teilaufgabe}
\begin{loesung}
    \punkte{Abgenutzte Messflächen am Messschieber.}{1}{} 
    \punkte{Abweichende mm-Teilung an der Skala eines Messschiebers}{1}{}
\end{loesung}
