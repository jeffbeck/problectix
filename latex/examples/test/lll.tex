                   \> \lll[138]{} \\
\\

Text zu 2:   \> \lll[138][s]{nocheinmal option s} \\
Text zu 3:   \> \lll[138][r]{Rechtsbündig auf der Linie mit Option r} \\
Text zu 4:   \> \lll[138][l]{Linksbündig auf der Linie mit Option l} \\
Text zu 5:   \> \lll[138][c]{Zentriert auf der Linie mit Option c} \\
\end{tabbing}
